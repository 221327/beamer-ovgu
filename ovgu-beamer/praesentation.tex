	\begin{block}{Satz des Pythagoras:}
		\begin{equation}
			a^2 + b^2 = c^2 \label{eq:pythogoras}
		\end{equation}	\end{block}
	\begin{block}{Es folgt, dass:}
		\begin{align}
			a^2 &= c^2 - b^2 \\
			b^2 &= c^2 - a^2
		\end{align}
